415	  Future<void> _handleCancelOrReschedule(
416	    Subject subject,
417	    DateTime date,
418	    int slotIndex,
419	    LectureSlot slot,
420	  ) async {
421	    final newDate = await showDatePicker(
422	      context: context,
423	      initialDate: date,
424	      firstDate: date,
425	      lastDate: DateTime(date.year + 1),
426	    );
427	
428	    if (newDate != null) {
429	      if (!mounted) return;
430	      final newStart = await showTimePicker(
431	        context: context,
432	        initialTime: slot.start,
433	      );
434	
435	      if (newStart != null) {
436	        final durationMinutes = _durationInMinutes(slot.start, slot.end);
437	        final newEnd = _addMinutes(newStart, durationMinutes);
438	        final newSlot = LectureSlot(start: newStart, end: newEnd);
439	
440	        _homeController.rescheduleAndCancel(
441	          subject,
442	          date,
443	          slotIndex,
444	          newDate,
445	          newSlot,
446	        );
447	        return;
448	      }
449	    }
450	
451	    _homeController.cancelLecture(subject, date, slotIndex);
452	    _showUndoMark(subject, date, slotIndex, null);
453	  }

475	  _StatusData _getStatusData({
476	    required AttendanceStatus? status,
477	    required DateTime slotDate,
478	    required int originalWeekday,
479	  }) {
480	    final isRescheduled = slotDate.weekday != originalWeekday;
481	    switch (status) {
482	      case AttendanceStatus.present:
483	        return _StatusData(
484	          isRescheduled ? "Present (Rescheduled)" : "Present",
485	          Colors.green,
486	        );
487	      case AttendanceStatus.absent:
488	        return _StatusData(
489	          isRescheduled ? "Absent (Rescheduled)" : "Absent",
490	          Colors.red,
491	        );
492	      case AttendanceStatus.cancelled:
493	        return _StatusData(
494	          isRescheduled ? "Cancelled (Rescheduled)" : "Cancelled",
495	          Colors.orange,
496	        );
497	      default:
498	        return _StatusData(
499	          isRescheduled ? "Not marked (Rescheduled)" : "Not marked",
500	          Colors.grey,
501	        );
502	    }
503	  }

505	  int _durationInMinutes(TimeOfDay start, TimeOfDay end) {
506	    final startMinutes = start.hour * 60 + start.minute;
507	    final endMinutes = end.hour * 60 + end.minute;
508	    final delta = (endMinutes - startMinutes) % (24 * 60);
509	    return delta == 0 ? 24 * 60 : delta;
510	  }
511	
512	  TimeOfDay _addMinutes(TimeOfDay start, int minutes) {
513	    final total = (start.hour * 60 + start.minute + minutes) % (24 * 60);
514	    return TimeOfDay(hour: total ~/ 60, minute: total % 60);
515	  }
